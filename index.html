Google Sheets Integration

Easy Setup:

  1. Create a new Google Sheet
  2. Go to Extensions → Apps Script in your Google Sheet
  3. Delete any existing code and paste this script:
function doPost(e) { const sheet = SpreadsheetApp.getActiveSheet(); const data = JSON.parse(e.postData.contents); // Clear existing data sheet.clear(); // Add headers const headers = ['Date', 'Category', 'Project/Client', 'Description', 'Duration']; sheet.getRange(1, 1, 1, headers.length).setValues([headers]); // Add data if (data.entries && data.entries.length > 0) { const rows = data.entries.map(entry => [ new Date(entry.date).toLocaleString(), entry.category, entry.project, entry.description || '', entry.formattedDuration ]); sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } return ContentService.createTextOutput('Success'); }
